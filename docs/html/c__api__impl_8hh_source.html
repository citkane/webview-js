47template <typename WorkFn, typename ResultFn>
48webview_error_t api_filter(WorkFn &&do_work, ResultFn &&put_result) noexcept {
49 try {
50 auto result = do_work();
51 if (result.ok()) {
52 put_result(result.value());
53 return WEBVIEW_ERROR_OK;
54 }
55 return result.error().code();
56 } catch (const exception &e) {
57 return e.error().code();
58 } catch (...) {
59 return WEBVIEW_ERROR_UNSPECIFIED;
60 }
61}

78inline webview *cast_to_webview(void *w) {
79 if (!w) {
80 throw exception{WEBVIEW_ERROR_INVALID_ARGUMENT,
81 "Cannot cast null pointer to webview instance"};
82 }
83 return static_cast<webview *>(w);
84}

89WEBVIEW_API webview_t webview_create(int debug, void *wnd) {
90 using namespace webview::detail;
91 webview::webview *w{};
92 auto err = api_filter(
93 [=]() -> webview::result<webview::webview *> {
94 return new webview::webview{static_cast<bool>(debug), wnd};
95 },
96 [&](webview::webview *w_) { w = w_; });
97 if (err == WEBVIEW_ERROR_OK) {
98 return w;
99 }
100 return nullptr;
101}

143WEBVIEW_API void *webview_get_native_handle(webview_t w,
144 webview_native_handle_kind_t kind) {
145 using namespace webview::detail;
146 void *handle{};
147 auto err = api_filter(
148 [=]() -> webview::result<void *> {
149 auto *w_ = cast_to_webview(w);
150 switch (kind) {
151 case WEBVIEW_NATIVE_HANDLE_KIND_UI_WINDOW:
152 return w_->window();
153 case WEBVIEW_NATIVE_HANDLE_KIND_UI_WIDGET:
154 return w_->widget();
155 case WEBVIEW_NATIVE_HANDLE_KIND_BROWSER_CONTROLLER:
156 return w_->browser_controller();
157 default:
158 return webview::error_info{WEBVIEW_ERROR_INVALID_ARGUMENT};
159 }
160 },
161 [&](void *handle_) { handle = handle_; });
162 if (err == WEBVIEW_ERROR_OK) {
163 return handle;
164 }
165 return nullptr;
166}
